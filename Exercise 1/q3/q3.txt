void initialize ( double  *A, int  N)
{
    #pragma omp parallel
    {
        #pragma omp single
        {
            for (int i = 0; i < N; i += 2) 
            {
                #pragma omp task firstprivate(i)
                {
                    A[i] = work(i);
                    if (i + 1 < N) 
                        A[i+1] = work(i+1);;
                }
            }
        }
    }
}
